#include<iostream>
#include <string>
#ifndef PERSONA_CPP
#define PERSONA_CPP
using namespace std;
class Persona{
 private:
   double peso;
 protected:
   string nombre;
 public:
   int edad;
  Persona(){
     edad=1;
     peso=1;
     nombre="Hola";
  }
  Persona(int valor,double decimal,string cadena){
      edad=valor;
      peso=decimal;
      nombre=cadena;
 }
};
#endif
#include <iostream>
include"Persona"
using namespace std
int main(){
 return 0;
}
